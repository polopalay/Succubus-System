random_yearly_everyone_pulse = {
	on_actions = {
		random_ntr_on_action
		sbs_devil_on_action
		hidden_effect_on_action
		sbs_seed_of_destruction_on_action
	}
}

random_ntr_on_action = {
	trigger = {
		is_adult = yes
		is_female = yes
		NOT = {
			has_trait = saint
			has_trait = chaste
			is_immortal = yes
		}
		NOT = { has_global_variable = seal_the_gates_of_hell }
		NOT = { faith.religion = religion:satanic_religion }
		exists = primary_spouse
		exists = primary_spouse.father
		primary_spouse = {
			NOT = { faith.religion = religion:satanic_religion }
			NOT = {
				has_trait = sbs_demon
				has_trait = seed_of_destruction
			}
		}
		primary_spouse.father = {
			is_incapable = no
			is_immortal = no
			is_ruler = yes
			is_landed = yes
			is_alive = yes
			NOT = { faith.religion = religion:satanic_religion }
			OR = {
				AND = {
					is_vassal_or_below_of = root
					root = { primary_title.tier >= tier_kingdom }
				}
				primary_title.tier >= tier_kingdom
				root = { is_ai = no }
				root = character:7757
				root = character:223522
			}
			OR = {
				is_vassal_of = root
				has_trait = lustful
				has_trait = ambitious
				has_trait = deceitful
				has_trait = greedy
				has_trait = cynical
				has_trait = seducer
				has_trait = rakish
				has_relation_lover = root
				root = { is_ai = no }
				root = character:7757
				root = character:223522
			}
			OR = {
				is_vassal_of = root
				has_relation_lover = root
				root.primary_spouse = {
					OR = {
						has_trait = albino
						has_trait = dwarf
						has_trait = infertile
						has_trait = weak
						has_trait = dull
						has_trait = beauty_bad
						has_trait = intellect_bad
						has_trait = physique_bad
					}
				}
				root = { is_ai = no }
				root = character:7757
				root = character:223522
			}
			OR = {
				NOT = {
					has_trait = chaste
					has_trait = content
					has_trait = honest
					has_trait = brave
					has_trait = zealous
					has_trait = saint
				}
				root = { is_ai = no }
				root = character:7757
				root = character:223522
			}
		}
	}
	random_events = {
		25 = sbs_ntr_event.0100
		10 = sbs_ntr_event.0210
		10 = sbs_ntr_event.0220
		10 = sbs_ntr_event.0230
		10 = sbs_ntr_event.0240
		10 = sbs_ntr_event.0250
	}
}

hidden_effect_on_action = {
	events = {
		# return slave to the owner
		carn_slave.004
		# saint is always protected by god
		sbs_event.9998
		# hof of satanic religion is always need spouse
		sbs_event.9999
	}
}

sbs_devil_on_action = {
	trigger = {
		is_adult = yes
		is_female = yes
		OR = {
			has_trait = sbs_succubus
			has_trait = sbs_lesser_succubus
		}
		is_imprisoned = no
	}
	random_events = {
		chance_to_happen = 50
		100 = sbs_devil.0550 # devil artifact
		50 = sbs_devil.1300 # special dream
		25 = sbs_devil.1410 # gremlin fuck
	}
}

sbs_seed_of_destruction_on_action = {
	trigger = {
		is_adult = yes
		is_female = yes
		is_lowborn = no
		has_trait = slave
		liege = { is_adult = yes }
		OR = {
			is_ai = no
			liege = { has_trait = seed_of_destruction }
		}
	}
	events = {
		sbs_slavery_event.0000
	}
}
